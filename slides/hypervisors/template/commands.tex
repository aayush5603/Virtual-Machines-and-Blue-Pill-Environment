% Final slide
% -----------
% Adds a slide with the specified text. It does not show any header or footer.
\newcommand{\finalslide}[1]{%
  \part{} % no section title in footer for this slides
  \begin{frame}[noframenumbering, plain]
    \begin{center}
      \vfill
      \Huge #1
      \vfill
    \end{center}
  \end{frame}
}
